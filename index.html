Developer API

Network Monitoring API

Integrate real-time network data into your applications

api.js 
// Initialize Toh Networks API
const tohAPI = new TohNetworksAPI({
    apiKey: 'your_api_key_here',
    environment: 'production'
});

// Get real-time network status
tohAPI.getNetworkStatus()
    .then(data => {
        console.log('Uptime:', data.uptime);
        console.log('Active Connections:', data.connections);
        console.log('Average Speed:', data.speed + 'Mbps');
    });

// Monitor specific service
tohAPI.monitorService('internet', {
    interval: 5000,
    callback: (status) => {
        updateDashboard(status);
    }
});
